outlier <- function(x) {
    outliers = quantile(x, c(0.25, 0.75))
    farness = 1.5 * (outliers[2] - outliers[1])
    return(c(x < (outliers[1] - farness) | x > (outliers[2] + farness)))
}
outlier(c(1, 4, 4, 7))

lettersMatch <- function(words, pattern) {
    containsPattern = c(grep(pattern, words))
    return(words[containsPattern])
}
lettersMatch(english, "o")

piSeries <- function(n) {
    x <- c()
    for (n in 1:n) {
        x <- c(x, (((-1)^(n - 1)) * (1 + (2 * (n - 1)))))
    }
    return(sum(4 * (1/(x))))
}
piSeries(600)

## [1] 3.14

howCloseToPi <- function(n) {
    PiApproximations <- c()
    for (p in 1:n) {
        series <- cumsum(piSeries(p))
        PiApproximations <- c(PiApproximations, series)
    }
    plot(PiApproximations, pch = 20, col = rgb(0, 0, 0, 0.4), log = "x")
}
howCloseToPi(100)
